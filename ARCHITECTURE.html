1 ภาพรวม & หลักคิด

กฎเหล็กของทั้งระบบ: Server เป็นเจ้าของความจริง (authoritative) — client แค่วาดหน้าจอกับส่ง "ความตั้งใจ"

🖥️ Browser (React)วาด UI · จับคลิก · emit event
เก็บแค่ "สำเนา" ของ state
⟷
WebSocket
2 ทาง
⚙️ Node Serverเก็บ state จริงในหน่วยความจำ
ตัดสินทุกกฎ · เก็บความลับ

ทำไมต้องให้ server ตัดสินทุกอย่าง?

  • กันโกง — ถ้า client รู้บทบาทคนอื่น จะเปิด DevTools โกงได้ทันที
  • ความลับ — server ส่งข้อมูลให้แต่ละคนไม่เท่ากัน (คุณเห็นแค่บทบาทตัวเอง)
  • ตัวกลางเดียว — ทุกคนเห็นสถานะตรงกัน ไม่มีการคำนวณซ้ำที่ client

วงจรพื้นฐาน (ทำซ้ำตลอดเกม)

คลิก→emit event→server คิด
↓
เปลี่ยน state→push state ใหม่→React วาดใหม่

3 โครงสร้างข้อมูล

ทุกอย่างในเกมคือ object GameRoom หนึ่งตัวต่อหนึ่งห้อง เก็บใน Map ในหน่วยความจำ server

GameRoom — state ทั้งหมดของห้อง

// gameLogic.js — constructor
class GameRoom {
  roomCode          // รหัสห้อง 4 ตัว เช่น "A3F9"
  players[]         // อาเรย์ผู้เล่น (ดูโครงด้านล่าง)
  phase             // เฟสปัจจุบัน: LOBBY / NIGHT / DAY ...
  settings          // { timer, enableFool, enableHunter, enableWitch, ... }

  // ── กลางคืน ──
  nightActions      // { werewolf:[], seer, doctor, cupid:[], witchHeal, witchPoison }
  nightQueue[]      // ลำดับบทบาทที่จะปลุกคืนนี้ เช่น ['Werewolf','Seer','Doctor']
  nightStepIndex    // ตอนนี้อยู่ step ที่เท่าไหร่ในคิว
  currentNightRole  // บทบาทที่กำลังทำ (ส่งให้ทุกคนเห็น)
  nightWerewolfVictim // เหยื่อหมาป่า (คำนวณไว้ให้แม่มดดู)
  pendingWitchKill  // เหยื่อที่แม่มดเห็น (ส่งให้แม่มดคนเดียว)

  // ── โหวต / ตาย / ชนะ ──
  votes             // Map: voterId → targetId
  lastVoteBreakdown // [{voter, target}] ไว้เผยผลโหวต
  lastNightKilled[] // id คนที่ตายเมื่อคืน
  winner            // null / 'VILLAGERS' / 'WEREWOLVES' / 'FOOL' / 'LOVERS'

  // ── บทบาทพิเศษ ──
  lovers[]          // [id, id] คู่รักจากคิวปิด
  cupidActed        // คิวปิดจับคู่แล้วหรือยัง
  witchPotions      // { heal:true, poison:true } ยาเหลือกี่ขวด
  pendingHunter     // id นายพรานที่กำลังจะยิง
  hunterTargetPhase // กลับไป DAY หรือ NIGHT หลังยิง

  // ── อื่นๆ ──
  dayNumber         // คืน/วันที่เท่าไหร่ (นับเพิ่มทุกคืน)
  messages[]        // แชท { senderId, text, type }
  gameLog[]         // บันทึกเหตุการณ์สาธารณะ
}

Player object

{

  socketId // id การเชื่อมต่อ (null=หลุด)
  name
  isHost
  role     // 'Werewolf' | 'Seer' | ...
  isAlive

}

ผลลัพธ์จาก getState(playerId) — กรองแล้ว เฉพาะคนถาม:

{

4 Backend — เริ่มทำงานยังไง

จุดเริ่มต้นสุด: บรรทัดล่างสุดของ server.js

// server.js — จุดเริ่มโปรแกรม
const io = new Server(httpServer)         // สร้าง Socket.IO
const rooms = new Map()                 // เก็บทุกห้อง: roomCode → GameRoom

io.on('connection', (socket) => {      // ★ มีคนต่อเข้ามา
   socket.on('create_room', ...)         // ลงทะเบียน handler ทุก event
   socket.on('night_action', ...)
   ... // (ดูตารางข้อ 5)
})

httpServer.listen(3001)              // ★★ เริ่มรอการเชื่อมต่อ

ลำดับการทำงานของ 1 event

↓

↓

6 Phase State Machine

หัวใจของทั้งเกม — room.phase บอกว่าตอนนี้อยู่ช่วงไหน ทั้ง backend และ frontend ตัดสินใจจากค่านี้

LOBBY→ ROLE_VIEW→ NIGHT→ NIGHT_WITCH→ DAY→ VOTING ↺ END_GAME
phaseอะไรพาเข้ามาอะไรพาออก
LOBBYสร้างห้องหัวหน้ากด start_game (≥4 คน)
ROLE_VIEWstartGame()อัตโนมัติหลัง 5 วิ → beginNight()
NIGHTbeginNight()ทำครบทุก step ในคิว
NIGHT_WITCHคิวถึง step แม่มดแม่มดใช้ยา / หมดเวลา
DAYfinalizeNight()15 วิ หรือหัวหน้ากดโหวต
VOTINGDAY จบทุกคนโหวตครบ → วนกลับ NIGHT
HUNTER_REVENGEนายพรานตาย (แทรกได้ทุกเมื่อ)นายพรานยิง → กลับ DAY/NIGHT
END_GAMEcheckWinCondition() เจอผู้ชนะหัวหน้ากด play_again → LOBBY
ตัวกลางเปลี่ยนเฟสหลักคือ processPhaseTransition() ใน server.js — เช็กเงื่อนไขชนะ, ส่ง night_result, ตั้งเวลาเฟสถัดไป

7 เครื่องยนต์กลางคืน (ทีละบทบาท)

ส่วนที่ซับซ้อนสุด — กลางคืนปลุกทีละบทบาทตามลำดับ ไม่ทำพร้อมกัน

💘 Cupid→ 🐺 Werewolf→ 🔮 Seer→ 💉 Doctor→ 🧙 Witch (ข้ามบทบาทที่ไม่มี/ตายแล้ว) 
// gameLogic.js — วงจรกลางคืน
beginNight() {
  setPhase('NIGHT')                  // dayNumber++, ล้าง nightActions
  nightQueue = buildNightQueue()     // สร้างลำดับเฉพาะบทบาทที่มี+ยังไม่ตาย
  nightStepIndex = 0
  startNightStep()                  // เปิด step แรก
}

startNightStep() {
  if (index >= queue.length) return finalizeNight()  // หมดคิว → จบ
  currentNightRole = queue[index]     // ★ ตั้งบทบาทที่ทำ (ทุกคนเห็น)
  if (role === 'Witch')
    pendingWitchKill = computeWerewolfVictim()  // แม่มดต้องเห็นเหยื่อก่อน
}

🖥️ Client (หมาป่า)
⚙️ Server
night_action { targetId }
1คลิกเหยื่อ + ยืนยัน
2handleNightAction() บันทึกลง nightActions.werewolf
3checkStepEnd()? หมาป่าครบทุกคน → advanceNightStep()
4step ต่อไป = Seer → continueNight()
game_state_update (currentNightRole='Seer')

// จบคิว → รวมผลทั้งคืนทีเดียว
finalizeNight() {
  if (cupid จับคู่) lovers = [...]          // 1. จับคู่รักก่อน (สำคัญ!)
  victim = werewolfVictim
  if (witchHeal) victim = null          // 2. แม่มดชุบ → รอด
  if (victim) killPlayer(victim)        // 3. ลงดาบหมาป่า

8 การตาย & ผลข้างเคียง

การตายทุกกรณีต้องผ่าน killPlayer(id, deathPhase) — เพราะมันจุดชนวนกลไกพิเศษ

// gameLogic.js — killPlayer
killPlayer(id, deathPhase) {
  player.isAlive = false

  if (role === 'Fool' && deathPhase === 'VOTING')
     winner = 'FOOL'                    // คนบ้าถูกโหวต = ชนะ!

  if (role === 'Hunter') {
     setPhase('HUNTER_REVENGE')         // นายพรานได้ยิงก่อนตาย
     pendingHunter = id
  }

  if (lovers.includes(id)) {            // คู่รักตรอมใจตายตาม
     killPlayer(otherLover, deathPhase)  // ← recursive!
  }
}

🤡 Fool

ถูกโหวตออก → winner='FOOL' ชนะทันที

🏹 Hunter

ตาย → เข้า HUNTER_REVENGE เลือกยิง 1 คน

💘 Lovers

คนหนึ่งตาย → อีกคนตายตาม (เรียกตัวเองซ้ำ)

เงื่อนไขชนะ — checkWinCondition()

ผลเงื่อนไข
VILLAGERSหมาป่าเหลือ 0 ตัว
WEREWOLVESหมาป่า ≥ ชาวบ้านที่เหลือ
FOOLคนบ้าถูกโหวตออก
LOVERSเหลือ 2 คนสุดท้ายเป็นคู่รักกัน

9 ระบบความลับ — getState(playerId)

จุดที่ทำให้เกมโกงไม่ได้ — server สร้าง state เฉพาะคนถาม ไม่เคยส่งข้อมูลลับให้ทุกคนเท่ากัน

// gameLogic.js — getState ตัดข้อมูลก่อนส่ง
getState(playerId) {
  players.map(p => ({
     ...p,
     role: canSee(playerId, p) ? p.role : null   // ★ ซ่อน role!
  }))
  // canSee = จริง เมื่อ: เป็นตัวเอง | หมาป่าดูหมาป่า | จบเกมแล้ว
}

กฎการมองเห็น

  • เห็น role ตัวเอง เท่านั้น
  • หมาป่า เห็นหมาป่าด้วยกัน
  • จบเกม (END_GAME) เห็นทุก role

กรองแชต 3 ช่อง

  • GLOBAL — ทุกคนเห็น
  • WEREWOLF — เฉพาะหมาป่า (เด็กน้อยแอบเห็น แต่ชื่อถูกซ่อน)
  • DEAD — เฉพาะคนตาย
เพราะ state ของแต่ละคนไม่เหมือนกัน → broadcastGameState() จึงต้องวน loop เรียก getState(p.id) ทีละคน แล้ว emit เข้า socket ของคนนั้น ไม่ใช่ส่งก้อนเดียวเข้าห้อง (io.to(room))

10 Frontend — เริ่มทำงานยังไง

index.html → main.jsx → mount <App/> — ทุกอย่างอยู่ใน App.jsx

// App.jsx — โครงหลัก
const socket = io(SERVER_URL)          // ★ สร้าง 1 ครั้ง (นอก component)

function App() {
  const [gameState, setGameState] = useState()   // สำเนา state จาก server
  const [playerId, setPlayerId] = useState()

  useEffect(() => {                    // ★ ลงทะเบียนหูฟังตอน mount
     socket.on('game_state_update', setGameState)   // หัวใจ!
     socket.on('seer_result', setSeerResult)
     socket.on('timer_update', setTimeRemaining)
     ...
  }, [])

  return renderPhase()               // เลือกจอตาม gameState.phase
}

renderPhase() — switch เลือกจอ

switch (gameState.phase) {
   case 'LOBBY':   return <Lobby/>
   case 'NIGHT':   return <NightPhase/>
   case 'VOTING':  return <VotingPhase/>
   ... // 1 phase = 1 component
}
💡 ทำไมแค่ setGameState แล้วจอเปลี่ยนเอง? เพราะ React: พอ state เปลี่ยน → component render ใหม่ → renderPhase() ถูกเรียกใหม่ → เลือก component ตาม phase ล่าสุด ทุกอย่างขับด้วย gameState ตัวเดียว

ทิศทางข้อมูล (วงจรปิด)

ผู้เล่นคลิก→handler เช่น handleVote()→socket.emit(...)
↓  (server ประมวลผล — ข้อ 4-8)
socket.on('game_state_update')→setGameState()→re-render

11 ตัวอย่างเต็ม: 1 คืน → 1 วัน

ไล่ตั้งแต่หัวหน้ากดเริ่มเกม จนได้ผู้ชนะ

  1. หัวหน้ากด "เริ่มเกม" → emit start_game → startGame() สุ่มบทบาท → phase = ROLE_VIEW → ทุกคนเห็นบทบาทตัวเอง 5 วิ
  2. อัตโนมัติ → beginNight() → สร้างคิว [Werewolf, Seer, Doctor] → phase NIGHT, currentNightRole = Werewolf
  3. 🐺 หมาป่า เลือกเหยื่อ → handleNightAction → step ต่อไป (Seer) — คนอื่นเห็น "หมาป่ากำลังทำหน้าที่"
  4. 🔮 ผู้หยั่งรู้ ตรวจ 1 คน → server emit seer_result เฉพาะผู้หยั่งรู้ → เห็น "X คือ หมาป่า"
  5. 💉 หมอ เลือกคุ้มครอง → หมดคิว → finalizeNight() ลงดาบ (เว้นถ้าหมอเซฟ) → phase DAY
  6. ☀️ กลางวัน → emit night_result → DayPhase ประกาศคนตาย → รอ 15 วิ หรือหัวหน้ากดโหวต
  7. 🗳️ โหวต → ทุกคน emit vote → ครบ → resolveVoting() ประหารคนคะแนนสูงสุด
  8. เช็กผู้ชนะ → checkWinCondition(): ยังไม่จบ → beginNight() วนคืนใหม่ · จบ → phase END_GAME เผยบทบาททุกคน

12 ระบบจับเวลา

แต่ละ step/phase มีตัวนับถอยหลัง ถ้าหมดเวลา = บังคับไปต่อ

// server.js
startRoomTimer(roomCode) {
   let timeLeft = room.settings.timer
   setInterval(() => {
      timeLeft--
      io.to(roomCode).emit('timer_update', timeLeft)   // อัปเดตแถบเวลา
      if (timeLeft <= 0) {                          // ★ หมดเวลา = บังคับไปต่อ
         if (phase==='NIGHT') advanceNightStep()   // ข้าม step
         if (phase==='VOTING') resolveVoting()    // จบโหวต
      }
   }, 1000)
}
roomTimersเก็บ setInterval ของแต่ละห้อง (clear ได้) roomTimeLeftเก็บวินาทีปัจจุบัน — ส่งให้คนที่รีเฟรชเห็นเวลาต่อทันที stopRoomTimerหยุด + ส่ง timer_update(null) ให้แถบเวลาหาย
ตอนคนกดครบก่อนหมดเวลา (เช่น หมาป่าเลือกเสร็จ) จะเรียก stopRoomTimer แล้วไป step ถัดไปทันที ไม่ต้องรอเวลาหมด
